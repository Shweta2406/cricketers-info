    {
       int coun;
       int num_batsman=0;
      
      country_code();
      printf("\nEnter the country code for which number of batsman are to be displayed: ");

    //2. Batsman with highest average score.
    void average_Score(struct cric_player cp[], int num)
    {
      int max,k;
      
      max=cp[0].batt_score;
      
      for(i=0; i<num; i++)
        {
          if(cp[i].category==1)

            if(cp[i].batt_score>cp[0].batt_score)

        }

  }
